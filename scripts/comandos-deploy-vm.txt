# Comandos para executar na VM (copiar e colar)

# PASSO 1: Fazer pull do código mais recente
git pull origin main

# PASSO 2: Parar os containers
docker compose down

# PASSO 3: Rebuild SEM CACHE (força usar código novo)
docker compose build --no-cache

# PASSO 4: Subir os containers
docker compose up -d

# PASSO 5: Ver os logs
docker compose logs -f rss-skull-bot

# EXPRESS: Tudo em um comando
git pull origin main && docker compose down && docker compose build --no-cache && docker compose up -d && docker compose logs -f rss-skull-bot
